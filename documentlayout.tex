% ------------------------------------------------------------------------------
% Typeface
% ------------------------------------------------------------------------------

\setupfontfeature[default]
  [onum=yes]

\definetypeface[newlib][rm][serif][libertine]
\definetypeface[newlib][ss][sans][biolinum]
\definetypeface[newlib][mm][math][times]

\setupbodyfont[newlib,12pt]
\setupbodyfontenvironment[default]
  [em=italic]

\setupcapitals
  [sc=yes]
\define[1]\cap
  {{\sc #1}}

% ------------------------------------------------------------------------------
% Pagelayout
% ------------------------------------------------------------------------------

\setuppapersize[A4][A4]

\setuplayout
  [width=middle,
   margin=6pc,
   height=middle,
   header=0pc,
   topspace=6pc,
   footer=4pc,
   bottomspace=6pc]

\startnotmode[doublesided]
  \setuplayout
    [backspace=8pc,
     cutspace=8pc]
  \setuppagenumbering
    [location=footer]
\stopmode

\startmode[doublesided]
  \setuplayout
    [backspace=6pc,
     cutspace=10pc]
  \setuppagenumbering
    [alternative=doublesided,
     location={footer,right}]
\stopmode

\startsectionblockenvironment[frontpart]
  \setuppagenumber
    [numberconversion=romannumerals]
\stopsectionblockenvironment

\startsectionblockenvironment[bodypart]
  \resetpagenumber
\stopsectionblockenvironment

\startsectionblockenvironment[appendix]
  \setuphead[part]
    [number=no]
  \setupsection[section-2]
    [previousnumber=no]
\stopsectionblockenvironment

% ------------------------------------------------------------------------------
% Text, Notes and Sides
% ------------------------------------------------------------------------------

\setupindenting[yes,medium]

\setupblank[big]

\setupitemgroup[itemize][each]
  [packed]
  [headstyle=\it,
   indentnext=auto]

\setupsynonyms[abbreviation]
  [textstyle=\tf,
   synonymstyle=\sc\word]

\setupurl
  [style=\tt]

\setupnotations
  [alternative=left,
   numbercommand=]
\setupnotes
  [rulethickness=0.8pt]

\setupdelimitedtext[quotation]
  [spacebefore=big,
   leftmargin=1.5em,
   style=\it]
\setupmargindata
  [style={\itx\setupinterlinespace[line=2.4ex]}]
\setupmargindata
  [margintext]
  [location=outer,
   align=flushleft,
   stack=continue,
   dy=1ex]

\setupfillinrules
  [separator=:]

\ifdefined\setupannotation
  \setupannotation
    [alternative=command,
     number=yes,
     command=\doannotation]
    
  \define[2]\doannotation
    {\high{\placeannotationnumber}\margintext{#1 #2}}
\fi

% ------------------------------------------------------------------------------
% Titles

\usemodule[title]

\setuptitle
  [spacebefore={force,2*big},
   spacebetween=big,
   spaceafter={3*big},
   setups=titlelayout]
\setuptitle[title]
  [style=\ss\bfd]
\setuptitle[subtitle]
  [style=\ss\bfb]
\setuptitle[subject]
  [style=\bfa]

\startsetups[titlelayout]
  \setupindenting[no]

  \blank[\roottitleparameter{spacebefore}]

  \startalignment[\roottitleparameter{align}]

    \blank[\roottitleparameter{spacebetween}]

    \titleelement{title}

    \titleelement{subtitle}

    \blank[\roottitleparameter{spacebetween}]

    \titleelement{author}
    
    \blank[\roottitleparameter{spacebetween}]

    \titleelement{date}

  \stopalignment

  \blank[\roottitleparameter{spaceafter}]
